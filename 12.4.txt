SELECT I.DESCRIPTION, IL.NAME
FROM ISSUES I, ISSUESLISTS IL
WHERE IL.ID = I.ISSUELIST_ID;

SELECT I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I, USERS U
WHERE U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASK_NUMBER
FROM ISSUES I, USERS U
GROUP BY U.FIRSTNAME, U.LASTNAME
HAVING COUNT(*) >0;
